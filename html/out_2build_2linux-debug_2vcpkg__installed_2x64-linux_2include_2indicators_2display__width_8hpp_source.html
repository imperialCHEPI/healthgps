20 /*
21  * This is an implementation of wcwidth() and wcswidth() (defined in
22  * IEEE Std 1002.1-2001) for Unicode.
23  *
24  * http://www.opengroup.org/onlinepubs/007904975/functions/wcwidth.html
25  * http://www.opengroup.org/onlinepubs/007904975/functions/wcswidth.html
26  *
27  * In fixed-width output devices, Latin characters all occupy a single
28  * "cell" position of equal width, whereas ideographic CJK characters
29  * occupy two such cells. Interoperability between terminal-line
30  * applications and (teletype-style) character terminals using the
31  * UTF-8 encoding requires agreement on which character should advance
32  * the cursor by how many cell positions. No established formal
33  * standards exist at present on which Unicode character shall occupy
34  * how many cell positions on character terminals. These routines are
35  * a first attempt of defining such behavior based on simple rules
36  * applied to data provided by the Unicode Consortium.
37  *
38  * For some graphical characters, the Unicode standard explicitly
39  * defines a character-cell width via the definition of the East Asian
40  * FullWidth (F), Wide (W), Half-width (H), and Narrow (Na) classes.
41  * In all these cases, there is no ambiguity about which width a
42  * terminal shall use. For characters in the East Asian Ambiguous (A)
43  * class, the width choice depends purely on a preference of backward
44  * compatibility with either historic CJK or Western practice.
45  * Choosing single-width for these characters is easy to justify as
46  * the appropriate long-term solution, as the CJK practice of
47  * displaying these characters as double-width comes from historic
48  * implementation simplicity (8-bit encoded characters were displayed
49  * single-width and 16-bit ones double-width, even for Greek,
50  * Cyrillic, etc.) and not any typographic considerations.
51  *
52  * Much less clear is the choice of width for the Not East Asian
53  * (Neutral) class. Existing practice does not dictate a width for any
54  * of these characters. It would nevertheless make sense
55  * typographically to allocate two character cells to characters such
56  * as for instance EM SPACE or VOLUME INTEGRAL, which cannot be
57  * represented adequately with a single-width glyph. The following
58  * routines at present merely assign a single-cell width to all
59  * neutral characters, in the interest of simplicity. This is not
60  * entirely satisfactory and should be reconsidered before
61  * establishing a formal standard in this area. At the moment, the
62  * decision which Not East Asian (Neutral) characters should be
63  * represented by double-width glyphs cannot yet be answered by
64  * applying a simple rule from the Unicode database content. Setting
65  * up a proper standard for the behavior of UTF-8 character terminals
66  * will require a careful analysis not only of each Unicode character,
67  * but also of each presentation form, something the author of these
68  * routines has avoided to do so far.
69  *
70  * http://www.unicode.org/unicode/reports/tr11/
71  *
72  * Markus Kuhn -- 2007-05-26 (Unicode 5.0)
73  *
74  * Permission to use, copy, modify, and distribute this software
75  * for any purpose and without fee is hereby granted. The author
76  * disclaims all warranties with regard to this software.
77  *
78  * Latest version: http://www.cl.cam.ac.uk/~mgk25/ucs/wcwidth.c
79  */

86 /* auxiliary function for binary search in interval table */
87 static inline int bisearch(wchar_t ucs, const struct interval *table, int max) {
88  int min = 0;
89  int mid;
90 
91  if (ucs < table[0].first || ucs > table[max].last)
92  return 0;
93  while (max >= min) {
94  mid = (min + max) / 2;
95  if (ucs > table[mid].last)
96  min = mid + 1;
97  else if (ucs < table[mid].first)
98  max = mid - 1;
99  else
100  return 1;
101  }
102 
103  return 0;
104 }

106 /* The following two functions define the column width of an ISO 10646
107  * character as follows:
108  *
109  * - The null character (U+0000) has a column width of 0.
110  *
111  * - Other C0/C1 control characters and DEL will lead to a return
112  * value of -1.
113  *
114  * - Non-spacing and enclosing combining characters (general
115  * category code Mn or Me in the Unicode database) have a
116  * column width of 0.
117  *
118  * - SOFT HYPHEN (U+00AD) has a column width of 1.
119  *
120  * - Other format characters (general category code Cf in the Unicode
121  * database) and ZERO WIDTH SPACE (U+200B) have a column width of 0.
122  *
123  * - Hangul Jamo medial vowels and final consonants (U+1160-U+11FF)
124  * have a column width of 0.
125  *
126  * - Spacing characters in the East Asian Wide (W) or East Asian
127  * Full-width (F) category as defined in Unicode Technical
128  * Report #11 have a column width of 2.
129  *
130  * - All remaining characters (including all printable
131  * ISO 8859-1 and WGL4 characters, Unicode control characters,
