207 #define NLOHMANN_JSON_SERIALIZE_ENUM(ENUM_TYPE, ...) \
208  template<typename BasicJsonType> \
209  inline void to_json(BasicJsonType& j, const ENUM_TYPE& e) \
210  { \
211  static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
212  static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
213  auto it = std::find_if(std::begin(m), std::end(m), \
214  [e](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
215  { \
216  return ej_pair.first == e; \
217  }); \
218  j = ((it != std::end(m)) ? it : std::begin(m))->second; \
219  } \
220  template<typename BasicJsonType> \
221  inline void from_json(const BasicJsonType& j, ENUM_TYPE& e) \
222  { \
223  static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
224  static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
225  auto it = std::find_if(std::begin(m), std::end(m), \
226  [&j](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
227  { \
228  return ej_pair.second == j; \
229  }); \
230  e = ((it != std::end(m)) ? it : std::begin(m))->first; \
231  }
