879  inline void win_change_attributes(std::ostream& stream, int foreground, int background)
880  {
881  // yeah, i know.. it's ugly, it's windows.
882  static WORD defaultAttributes = 0;
883 
884  // Windows doesn't have ANSI escape sequences and so we use special
885  // API to change Terminal output color. That means we can't
886  // manipulate colors by means of "std::stringstream" and hence
887  // should do nothing in this case.
888  if (!_internal::is_atty(stream))
889  return;

893  if (&stream == &std::cout)

896  hTerminal = GetStdHandle(STD_ERROR_HANDLE);
897 
898  // save default terminal attributes if it unsaved
899  if (!defaultAttributes)
900  {
901  CONSOLE_SCREEN_BUFFER_INFO info;
902  if (!GetConsoleScreenBufferInfo(hTerminal, &info))
903  return;
904  defaultAttributes = info.wAttributes;
905  }
906 
907  // restore all default settings
908  if (foreground == -1 && background == -1)
909  {
910  SetConsoleTextAttribute(hTerminal, defaultAttributes);
911  return;
912  }
913 
914  // get current settings
915  CONSOLE_SCREEN_BUFFER_INFO info;
916  if (!GetConsoleScreenBufferInfo(hTerminal, &info))
917  return;
918

920  {

922  info.wAttributes |= static_cast<WORD>(foreground);
923  }
924 
925  if (background != -1)
926  {
927  info.wAttributes &= ~(info.wAttributes & 0xF0);
928  info.wAttributes |= static_cast<WORD>(background);

1359 /*
1360  * This is an implementation of wcwidth() and wcswidth() (defined in
1361  * IEEE Std 1002.1-2001) for Unicode.
1362  *
1363  * http://www.opengroup.org/onlinepubs/007904975/functions/wcwidth.html
1364  * http://www.opengroup.org/onlinepubs/007904975/functions/wcswidth.html
1365  *
1366  * In fixed-width output devices, Latin characters all occupy a single
1367  * "cell" position of equal width, whereas ideographic CJK characters
1368  * occupy two such cells. Interoperability between terminal-line
1369  * applications and (teletype-style) character terminals using the
1370  * UTF-8 encoding requires agreement on which character should advance
1371  * the cursor by how many cell positions. No established formal
1372  * standards exist at present on which Unicode character shall occupy
1373  * how many cell positions on character terminals. These routines are
1374  * a first attempt of defining such behavior based on simple rules
1375  * applied to data provided by the Unicode Consortium.
1376  *
1377  * For some graphical characters, the Unicode standard explicitly
1378  * defines a character-cell width via the definition of the East Asian
1379  * FullWidth (F), Wide (W), Half-width (H), and Narrow (Na) classes.
1380  * In all these cases, there is no ambiguity about which width a
1381  * terminal shall use. For characters in the East Asian Ambiguous (A)
1382  * class, the width choice depends purely on a preference of backward
1383  * compatibility with either historic CJK or Western practice.
1384  * Choosing single-width for these characters is easy to justify as
1385  * the appropriate long-term solution, as the CJK practice of
1386  * displaying these characters as double-width comes from historic
1387  * implementation simplicity (8-bit encoded characters were displayed
1388  * single-width and 16-bit ones double-width, even for Greek,
1389  * Cyrillic, etc.) and not any typographic considerations.
1390  *
1391  * Much less clear is the choice of width for the Not East Asian
1392  * (Neutral) class. Existing practice does not dictate a width for any
1393  * of these characters. It would nevertheless make sense
1394  * typographically to allocate two character cells to characters such
1395  * as for instance EM SPACE or VOLUME INTEGRAL, which cannot be
1396  * represented adequately with a single-width glyph. The following
1397  * routines at present merely assign a single-cell width to all
1398  * neutral characters, in the interest of simplicity. This is not
1399  * entirely satisfactory and should be reconsidered before
1400  * establishing a formal standard in this area. At the moment, the
1401  * decision which Not East Asian (Neutral) characters should be
1402  * represented by double-width glyphs cannot yet be answered by
1403  * applying a simple rule from the Unicode database content. Setting
1404  * up a proper standard for the behavior of UTF-8 character terminals
1405  * will require a careful analysis not only of each Unicode character,
1406  * but also of each presentation form, something the author of these
1407  * routines has avoided to do so far.
1408  *
1409  * http://www.unicode.org/unicode/reports/tr11/
1410  *
1411  * Markus Kuhn -- 2007-05-26 (Unicode 5.0)
1412  *
1413  * Permission to use, copy, modify, and distribute this software
1414  * for any purpose and without fee is hereby granted. The author
1415  * disclaims all warranties with regard to this software.
1416  *
1417  * Latest version: http://www.cl.cam.ac.uk/~mgk25/ucs/wcwidth.c

1426 static inline int bisearch(wchar_t ucs, const struct interval *table, int max) {
1427  int min = 0;
1428  int mid;
1429 
1430  if (ucs < table[0].first || ucs > table[max].last)
1431  return 0;
1432  while (max >= min) {
1433  mid = (min + max) / 2;
1434  if (ucs > table[mid].last)
1435  min = mid + 1;
1436  else if (ucs < table[mid].first)
1437  max = mid - 1;
1438  else
1439  return 1;
1440  }
1441 
1442  return 0;
1443 }
