134  void ToStr(const T& pVal, std::string& pStr) const
135  {
136  if (typeid(T) == typeid(int) ||
137  typeid(T) == typeid(long) ||
138  typeid(T) == typeid(long long) ||
139  typeid(T) == typeid(unsigned) ||
140  typeid(T) == typeid(unsigned long) ||
141  typeid(T) == typeid(unsigned long long) ||
142  typeid(T) == typeid(float) ||
143  typeid(T) == typeid(double) ||
144  typeid(T) == typeid(long double) ||
145  typeid(T) == typeid(char))
146  {
147  std::ostringstream out;

149  pStr = out.str();
150  }
151  else
152  {
153  throw no_converter();
154  }

162  void ToVal(const std::string& pStr, T& pVal) const
163  {
164  try
165  {
166  if (typeid(T) == typeid(int))
167  {
168  pVal = static_cast<T>(std::stoi(pStr));
169  return;
170  }
171  else if (typeid(T) == typeid(long))
172  {
173  pVal = static_cast<T>(std::stol(pStr));
174  return;
175  }
176  else if (typeid(T) == typeid(long long))
177  {
178  pVal = static_cast<T>(std::stoll(pStr));
179  return;
180  }
181  else if (typeid(T) == typeid(unsigned))
182  {
183  pVal = static_cast<T>(std::stoul(pStr));
184  return;
185  }
186  else if (typeid(T) == typeid(unsigned long))
187  {
188  pVal = static_cast<T>(std::stoul(pStr));
189  return;
190  }
191  else if (typeid(T) == typeid(unsigned long long))
192  {
193  pVal = static_cast<T>(std::stoull(pStr));
194  return;
195  }
196  }
197  catch (...)
198  {
199  if (!mConverterParams.mHasDefaultConverter)
200  {
201  throw;
202  }
203  else
204  {
205  pVal = static_cast<T>(mConverterParams.mDefaultInteger);
206  return;
207  }
208  }
209 
210  try
211  {
212  if (mConverterParams.mNumericLocale)
213  {
214  if (typeid(T) == typeid(float))
215  {
216  pVal = static_cast<T>(std::stof(pStr));
217  return;
218  }
219  else if (typeid(T) == typeid(double))
220  {
221  pVal = static_cast<T>(std::stod(pStr));
222  return;
223  }
224  else if (typeid(T) == typeid(long double))
225  {
226  pVal = static_cast<T>(std::stold(pStr));
227  return;
228  }
229  }
230  else
231  {
232  if ((typeid(T) == typeid(float)) ||
233  (typeid(T) == typeid(double)) ||
234  (typeid(T) == typeid(long double)))
235  {
236  std::istringstream iss(pStr);
237  iss >> pVal;
238  if (iss.fail() || iss.bad() || !iss.eof())
239  {
240  throw std::invalid_argument("istringstream: no conversion");
241  }
242  return;
243  }
244  }
245  }
246  catch (...)
247  {
248  if (!mConverterParams.mHasDefaultConverter)
249  {
250  throw;
251  }
252  else
253  {
254  pVal = static_cast<T>(mConverterParams.mDefaultFloat);
255  return;
256  }
257  }
258 
259  if (typeid(T) == typeid(char))
260  {
261  pVal = static_cast<T>(pStr[0]);
262  return;
263  }
264  else
265  {
266  throw no_converter();
267  }
268  }
269 
270  private:
271  const ConverterParams& mConverterParams;
272  };

588  template<typename T>
589  std::vector<T> GetColumn(const size_t pColumnIdx) const
590  {
591  const ssize_t columnIdx = pColumnIdx + (mLabelParams.mRowNameIdx + 1);
592  std::vector<T> column;
593  Converter<T> converter(mConverterParams);
594  for (auto itRow = mData.begin(); itRow != mData.end(); ++itRow)
595  {
596  if (std::distance(mData.begin(), itRow) > mLabelParams.mColumnNameIdx)
597  {
598  if (columnIdx < static_cast<ssize_t>(itRow->size()))
599  {
600  T val;
601  converter.ToVal(itRow->at(columnIdx), val);
602  column.push_back(val);
603  }
604  else
605  {
606  const std::string errStr = "requested column index " +
607  std::to_string(columnIdx - (mLabelParams.mRowNameIdx + 1)) + " >= " +
608  std::to_string(itRow->size() - (mLabelParams.mRowNameIdx + 1)) +
609  " (number of columns on row index " +
610  std::to_string(std::distance(mData.begin(), itRow) -
611  (mLabelParams.mColumnNameIdx + 1)) + ")";
612  throw std::out_of_range(errStr);
613  }
614  }
615  }
616  return column;
617  }
